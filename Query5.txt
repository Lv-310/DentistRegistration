IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLES 
           WHERE TABLE_NAME = 'SERVICES')
BEGIN
	CREATE TABLE SERVICES (
        ID_SERVICE INT NOT NULL,
	NAME_SERVICE NVARCHAR(50) NOT NULL,
        DESCRIPTION_SERVICE NVARCHAR(128) NOT NULL,
	CONSTRAINT PK_SERVICES PRIMARY KEY (ID_SERVICE))
END;

IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLES 
           WHERE TABLE_NAME = 'PRICES')
BEGIN
        CREATE TABLE PRICES (
        ID_PRICE INT NOT NULL,
	PRICE INT NOT NULL,
        DATE_START_PRICE DATE NOT NULL,
        DATE_END_PRICE DATE NOT NULL,
	SERVICE_ID INT NOT NULL,
	CONSTRAINT PK_PRICES PRIMARY KEY (ID_PRICE),
	CONSTRAINT FK_SERVICES FOREIGN KEY (SERVICE_ID), 
        REFERENCES SERVICES(ID_SERVICE))
END;
